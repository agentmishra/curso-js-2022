Clase 3

for, while,
do...while
y switch

Curso de JavaScript

for

for (let i = 1; i < 4; i++) {
    console.log("Iteración " + i);
}

while

let x = 1;
while (x < 4) {
    console.log("Iteración " + x);
    x++;
}

do...while

let y = 1;
do {
    console.log("Iteración " + y);
    y++;
} while (y < 4);

switch

let z = 1;
switch (z) {
    case 1:
        console.log("Z es igual a 1");
        break;
    case 2:
        console.log("Z es igual a 2");
        break;
    case 3:
        console.log("Z es igual a 3");
        break;
    default:
        console.log("No sé el valor de Z");
        break;
}
